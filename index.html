function doPost(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName('가계부'); if (!sh) { sh = ss.insertSheet('가계부'); sh.appendRow(['날짜','유형','작성자','카테고리','카드','내용','금액']); } var d = JSON.parse(e.postData.contents); var writer = d.writer==='hub'?'남편':d.writer==='wife'?'와이프':'공용'; sh.appendRow([d.date, d.type==='income'?'수입':'지출', writer, d.cat, d.card||'', d.memo, d.amount]); return ContentService .createTextOutput(JSON.stringify({result:'ok'})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({error:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
